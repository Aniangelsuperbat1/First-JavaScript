-Three parts to a loop. Initizlization, condition, update. ICU.

While Loop:
let i = 1; #initialization
while(i <11){ #condition
    #code
    console.log(i);
    i++; #update
}

do-while loop:
let i = 1000; #initialization
do{
    console.log(i); #code
    i++; #update
    while(i < 10); #condition
}

for loop:
for (let i=0; #initialization i < 10; #condition i++ #update) {
    console.log(i); #code
}
